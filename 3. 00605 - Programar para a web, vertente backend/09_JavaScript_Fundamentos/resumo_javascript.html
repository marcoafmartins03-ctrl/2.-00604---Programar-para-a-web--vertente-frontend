3. Condicionais

4. Loops

// for - repetir X vezes for (let i = 0; i < 5; i++) { console.log(i); } // for...of - percorrer array const frutas = ["maçã", "banana", "laranja"]; for (const fruta of frutas) { console.log(fruta); } // forEach - alternativa frutas.forEach(function(fruta) { console.log(fruta); });

5. Funções

// Função tradicional function somar(a, b) { return a + b; } // Arrow function (forma curta) const multiplicar = (a, b) => a * b; // Usar console.log(somar(2, 3)); // 5 console.log(multiplicar(2, 3)); // 6

6. Arrays

const numeros = [1, 2, 3, 4, 5]; numeros.length; // 5 (tamanho) numeros[0]; // 1 (primeiro) numeros.push(6); // adiciona no fim numeros.pop(); // remove do fim numeros.includes(3); // true (existe?) // Métodos úteis numeros.map(n => n * 2); // [2, 4, 6, 8, 10] numeros.filter(n => n > 2); // [3, 4, 5] numeros.find(n => n > 2); // 3 (primeiro)

7. Objetos

const pessoa = { nome: "Ana", idade: 25, cidade: "Lisboa", // Método (função no objeto) apresentar: function() { return "Olá, sou " + this.nome; } }; // Aceder propriedades pessoa.nome; // "Ana" pessoa["idade"]; // 25 // Adicionar/modificar pessoa.email = "ana@email.com"; pessoa.idade = 26;
